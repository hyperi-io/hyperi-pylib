  namespace: {namespace}

    - name: POD_NAME

    - name: config
      mountPath: /config
    - name: data
      mountPath: /data
  volumes:
  - name: app-code
    configMap:
      name: {configmap_name}
  - name: config
    emptyDir: {}
  - name: data
    emptyDir: {}
