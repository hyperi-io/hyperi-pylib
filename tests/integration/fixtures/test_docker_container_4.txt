import os
import json

# Simulate what hyperi-pylib would detect
mounts = {
    "config_dir": "/app/config" if os.path.exists("/app/config") else "/config",
    "data_dir": "/app/data" if os.path.exists("/app/data") else "/data",
    "secrets_dir": "/run/secrets" if os.path.exists("/run/secrets") else "/app/secrets",
    "logs_dir": "/app/logs" if os.path.exists("/app/logs") else "/logs",
    "temp_dir": "/tmp"
}

# Check which directories exist
for name, path in mounts.items():
    if os.path.exists(path):
        print(f"{name}: {path} EXISTS")
    else:
        print(f"{name}: {path} NOT_FOUND")
