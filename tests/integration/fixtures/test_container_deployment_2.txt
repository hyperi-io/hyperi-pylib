import sys
import os
sys.path.insert(0, '/app')

from hs_lib import Application, create_metrics
from hs_lib.prometheus import PrometheusMetrics
import time

# Create API application
app = Application.api(name="metrics-test-api", port=8000, metrics_port=8080)

# Create custom metrics
metrics = create_metrics(app_name="hyperlib_test")

# Add a counter metric
request_counter = metrics.counter(
    "requests_total",
    "Total number of requests",
    labels=["endpoint", "method"]
)

# Add a gauge metric
active_connections = metrics.gauge(
    "active_connections",
    "Number of active connections"
)

# Add a histogram metric
request_duration = metrics.histogram(
    "request_duration_seconds",
    "Request duration in seconds",
    buckets=[0.1, 0.25, 0.5, 1.0, 2.5, 5.0]
)

@app.route("/")
def root():
    # Increment counter
    request_counter.labels(endpoint="/", method="GET").inc()

    # Simulate some work
    start_time = time.time()
    time.sleep(0.1)
    duration = time.time() - start_time

    # Record duration
    request_duration.observe(duration)

    return {"status": "ok", "metrics": "enabled"}

@app.route("/connections")
def connections():
    # Update gauge
    import random
    active_connections.set(random.randint(1, 100))
    return {"connections": "updated"}

if __name__ == "__main__":
    # Start metrics server in background
    import threading
    from prometheus_client import start_http_server

    def start_metrics():
        start_http_server(8080)

    metrics_thread = threading.Thread(target=start_metrics, daemon=True)
    metrics_thread.start()

    # Run API
    app.run()
